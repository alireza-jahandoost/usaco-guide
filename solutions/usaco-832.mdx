## Solution 2

In this problem, there are three cases that we have to consider:
1. Cow $1$ is fixed: In this case, the solution is simple, as we just output its fixed position.
2. Cow $1$ is in the given hierarchy of cows: In this case, we want to place every cow in the hierarchy as early as possible.
3. Cow $1$ is neither fixed nor in the hierarchy: In this case, we would want to place every cow in the hierarchy as late as possible, such that cow $1$ gets the earliest possible position.

## Implementation

**Time Complexity:** $\mathcal{O}(N(M + N))$

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
	freopen("milkorder.in", "r", stdin);
	freopen("milkorder.out", "w", stdout);
	int n,m,k;
	cin>>n>>m>>k;

	vector<int> pos(n+1, -1);
	map<int,int> statuspos;
	vector<int> status(m);
	for(int i=0;i<m;i++){
		cin>>status[i];
		statuspos[status[i]] = i;
	}

	set<int> exist;
	for(int i=0;i<k;i++){
		int c,p;
		cin>>c>>p;
		pos[p] = c;
		exist.insert(c);
	}

		// Case 1
	for(int i=1;i<=n;i++)if(pos[i] == 1){
		cout<<i<<endl;
		return 0;
	}

		// Case 2
	if(statuspos.count(1)){
		int start = statuspos[1]-1;
		while(start>=0){
			if(exist.count(status[start]))break;
			start--;
		}
		bool check = start == -1;
		int diff = statuspos[1]-start;
		for(int i=1;i<=n;i++){
			if(!check&&pos[i]==status[start]){
				check = true;
				continue;
			}
			if(check&&pos[i]==-1)diff--;
			if(diff == 0){
				cout<<i<<endl;
				break;
			}
		}
		return 0;
	}

		// Case 3
	for(int i=1;i<=n;i++){
		if(pos[i] == -1)continue;
		if(statuspos.count(pos[i])){
			int a = statuspos[pos[i]]-1;
			int b = i-1;
			while(a>=0&&b>0){
				if(pos[b]!=-1){
					b--;
					continue;
				}
				if(exist.count(status[a])){
					a--;
					continue;
				}
				pos[b] = status[a];
				exist.insert(status[a]);
				b--;a--;
			}
		}
	}
	for(int i=1;i<=n;i++)if(pos[i] == -1){
		cout<<i<<endl;
		break;
	}

	return 0;
}
```

```py
file_in = open('milkorder.in')
data = file_in.read().strip().split('\n')
n, m, k = map(int, data[0].split(' '))
hierarchy = list(map(int, data[1].split(' ')))
fixed = [list(map(int, data[i + 2].split(' '))) for i in range(k)]

cows = [0 for i in range(n)]

for i in range(k):
  	cows[fixed[i][1] - 1] = fixed[i][0]

if 1 in cows:
	# Case 1
 	ans = cows.index(1) + 1
else:
	if 1 not in hierarchy:
		# Case 2
		i = n - 1
		pointer = m - 1
		while pointer >= 0:
			if hierarchy[pointer] in cows:
				i = cows.index(hierarchy[pointer]) - 1
				pointer -= 1
			else:
				if cows[i] == 0:
					cows[i] = hierarchy[pointer]
					i -= 1
					pointer -= 1
				else:
					i -= 1

		for i in range(n):
			if cows[i] == 0:
				ans = i + 1
				break
	else:
		# Case 3
		i = 0
		pointer = 0
		while pointer <= m - 1:
			if hierarchy[pointer] in cows:
				i = cows.index(hierarchy[pointer]) + 1
				pointer += 1
			else:
				if cows[i] == 0:
					cows[i] = hierarchy[pointer]
					i += 1
					pointer += 1
				else:
					i += 1
		for i in range(n):
			if cows[i] == 1:
				ans = i + 1
				break

print(ans, file=open('milkorder.out', 'w'))
```
